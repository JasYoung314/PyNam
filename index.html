import random  # Pseudo random number generation
# Simulate two people walking towards each other 10 times
K = 10
player_1 = [random.choice('LR') for k in range(K)]
player_2 = [random.choice('LR') for k in range(K)]
					

# Count the number of impacts
impacts = sum([player_1[k] == player_2[k] for k in range(K)])
					

# Let us make this repeatable
def walk(K):
    """
    Simulate people walking along the pavement K times
    """
    player_1 = [random.choice('LR') for k in range(K)]
    player_2 = [random.choice('LR') for k in range(K)]
    impacts = sum([player_1[k] == player_2[k] for k in range(K)])
    return impacts
					

from bashplotlib.histogram import plot_hist

                        showSummary=True)
